141 /*
142 Manages render targets, including the layer stack and postprocessing framebuffers.
143
144 Layers can be pushed and popped, creating new framebuffers as needed. Typically, geometry is rendered to the top
145 layer. The layer framebuffers may have MSAA enabled.
146
147 Postprocessing framebuffers are separate from the layers, and are commonly used to apply texture-wide effects
148 such as filters. They are used both as input and output during rendering, and do not use MSAA.
149 */
150 class RenderLayerStack {
151 public:
152 RenderLayerStack();
153 ~RenderLayerStack();
154
155 // Push a new layer. All references to previously retrieved layers are invalidated.
156 Rml::LayerHandle PushLayer();
157
158 // Pop the top layer. All references to previously retrieved layers are invalidated.
159 void PopLayer();
160
161 const Gfx::FramebufferData& GetLayer(Rml::LayerHandle layer) const;
162 const Gfx::FramebufferData& GetTopLayer() const;
163 Rml::LayerHandle GetTopLayerHandle() const;
164
165 const Gfx::FramebufferData& GetPostprocessPrimary() { return EnsureFramebufferPostprocess(0); }
166 const Gfx::FramebufferData& GetPostprocessSecondary() { return EnsureFramebufferPostprocess(1); }
167 const Gfx::FramebufferData& GetPostprocessTertiary() { return EnsureFramebufferPostprocess(2); }
168 const Gfx::FramebufferData& GetBlendMask() { return EnsureFramebufferPostprocess(3); }
169
170 void SwapPostprocessPrimarySecondary();
171
172 void BeginFrame(int new_width, int new_height);
173 void EndFrame();
174
175 private:
176 void DestroyFramebuffers();
177 const Gfx::FramebufferData& EnsureFramebufferPostprocess(int index);
178
179 int width = 0, height = 0;
180
181 // The number of active layers is manually tracked since we re-use the framebuffers stored in the fb_layers stack.
182 int layers_size = 0;
183
184 Rml::Vector<Gfx::FramebufferData> fb_layers;
185 Rml::Vector<Gfx::FramebufferData> fb_postprocess;
186 };
